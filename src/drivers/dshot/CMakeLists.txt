set(PARAM_PREFIX PWM_MAIN)
set(DSHOT_MODULE_PARAMS)
set(DSHOT_MODULE_SRCS)

if(CONFIG_BOARD_IO)
	set(PARAM_PREFIX PWM_AUX)
endif()

if(CONFIG_DSHOT_AM32_PROG)
	list(APPEND DSHOT_MODULE_PARAMS esc/am32.yaml)
	list(APPEND DSHOT_MODULE_SRCS esc/AM32Settings.cpp)
endif()

px4_add_module(
	MODULE drivers__dshot
	MAIN dshot
	COMPILE_FLAGS
		-DPARAM_PREFIX="${PARAM_PREFIX}"
		# -DDEBUG_BUILD
	SRCS
		DShot.cpp
		DShotTelemetry.cpp
		${DSHOT_MODULE_SRCS}
	DEPENDS
		arch_io_pins
		arch_dshot
		mixer_module
	MODULE_CONFIG
		module.yaml
		${DSHOT_MODULE_PARAMS}
	)
