	USES_TERMINAL
	DEPENDS
		px4
		${PX4_SOURCE_DIR}/ROMFS/px4fmu_common/init.d-posix/airframes/10016_none_iris
	COMMENT "launching px4 none_iris (SYS_AUTOSTART=10016)"
)
